Product Schema (with nested variants)

const productSchema = new Schema({
  name:        { type: String, required: true, minlength: 2 },
  category:    { type: String, required: true,
                 enum: ['Electronics','Clothing','Appliances','Sports','Home'] },
  brand:       { type: String, default: '' },
  price:       { type: Number, required: true, min: 0 },
  rating:      { type: Number, default: null, min: 0, max: 5 },
  description: { type: String, default: '' },
  variants:    [variantSchema],   // ← nested array
  inStock:     { type: Boolean, default: false },
}, { timestamps: true });

Pre-Save Middleware

productSchema.pre('save', function(next) {
  const totalStock = this.variants.reduce((sum, v) => sum + v.stock, 0);
  this.inStock = totalStock > 0;  // auto-set
  next();
});

const Product = mongoose.model('Product', productSchema);
module.exports = Product;

CRUD Operations

  variants: [{ sku: 'IPH-BLK', color: 'Black', stock: 20 }],
});
await product.save();

Aggregation Pipelines

How Aggregation Works

const result = await Product.aggregate([
  { $unwind: "$variants" },              // flatten array
  { $match: { "variants.stock": { $lt: 10 } } },  // filter
  { $group: {
      _id:       "$category",
      avgRating: { $avg: "$rating" },
      count:     { $sum: 1 },
  }},
  { $sort: { avgRating: -1 } },          // sort
]);
